%%%%%%%%%%%%%%%%%%% the chapter headings
%
\newcounter{seempChapter}
\newcommand{\seempSection}[1]{%
	\stepcounter{seempChapter}		
	{
		\par\noindent
		\vspace{5pt}
		\huge{\greenheavy{Chapter \theseempChapter: #1}}\par
		\vspace{3pt}
	}
	\addcontentsline{toc}{section}{
		\greenheavy{
			Chapter \theseempChapter: #1}
			}
}	%

%%%%%%%%%%%%%%%%%%% the section headings
%
\newcounter{seempsection}
\newcommand{\seempSubsection}[1]{%
	\stepcounter{seempsection}
	{
		\par\noindent
		\vspace{3pt}
		\Large{\blueheavy{\theseempsection. #1}}\par
		\vspace{3pt}
	}
	\addcontentsline{toc}{subsubsection}{
		\blueheavy{
			\theseempsection. #1}
			}
}%


%%%%%%%%%%%%%%%%% The gray headers

\newcounter{seempGrayHeaderCounter}
\newcommand{\seempGrayHeader}[1]{%
	\stepcounter{seempGrayHeaderCounter}	%
	{
		\par\noindent
		\vspace{3pt}
		\seempgray{\Large{\Alph{seempGrayHeaderCounter}. #1}}\par	
		\vspace{3pt}
	}%
	\addcontentsline{toc}{subsection}{
		\seempgray{
			\Alph{seempGrayHeaderCounter}. #1}
			} %
}%
